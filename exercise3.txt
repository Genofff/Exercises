// Додатково: Зробіть програму, яка може ділити, множити, та віднімати ці два номери

let firstNumber = +prompt("Введіть перше число:", []);
let secondNumber = +prompt("Введіть друге число:", []);

let result1 = firstNumber + secondNumber;
alert("Сума чисел: " + result1);

let result2 = Number(firstNumber) - Number(secondNumber);
alert("Віднімання чисел: " + result2);

let result3 = firstNumber * secondNumber;
alert("Множення чисел: " + result3);

let result4 = Number(firstNumber) / Number(secondNumber);
alert("Ділення чисел: " + result4);
